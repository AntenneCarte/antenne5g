  <!-- CSS Leaflet -->

  <link rel="stylesheet" href="https://unpkg.com/leaflet/dist/leaflet.css" />

  <style>

    #map { height: 100vh; width: 100%; }

  </style>

</head>

<div id="map"></div>


<!-- JS Leaflet -->

<script src="https://unpkg.com/leaflet/dist/leaflet.js"></script>


<script>

  // Initialiser la carte centrée sur la Suisse

  var map = L.map('map').setView([46.8, 8.2], 8);


  // Ajouter un fond OpenStreetMap

  L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {

    maxZoom: 18,

  }).addTo(map);


  // Fonction pour style des faisceaux selon fréquence (exemple)

  function style(feature) {

    var freq = feature.properties.frequency || "";

    var color = 'pink';

    if (freq.includes('900')) color = 'orange';

    else if (freq.includes('1000')) color = 'purple';

    else if (freq.includes('3000')) color = 'blue';

    return {

      fillColor: color,

      weight: 1,

      opacity: 1,

      color: color,

      fillOpacity: 0.4

    };

  }


  // Charger le GeoJSON des faisceaux depuis le même dossier

  fetch('antennas_beams.geojson')

    .then(response => response.json())

    .then(data => {

      L.geoJSON(data, {

        style: style,

        onEachFeature: function (feature, layer) {

          var freq = feature.properties.frequency || "N/A";

          var az = feature.properties.azimuth || "N/A";

          layer.bindPopup("Fréquence : " + freq + "<br>Azimut : " + az);

        }

      }).addTo(map);

    })

    .catch(err => console.error("Erreur chargement GeoJSON:", err));

</script>
